def on_press(direction, look = true)
	if !@activated
		@activated = true
		@action_manager.restart
	end
	case direction
		when Direction::TOP
			@direction = Direction::BOT
		when Direction::BOT
			@direction = Direction::TOP
		when Direction::LEF
			@direction = Direction::RIG
		when Direction::RIG
			@direction = Direction::LEF
		else
			puts "[ERROR] Unknown direction #{direction}"
	end if look
end
